**Lógica de implementação:**

```
main():
  1. parse_args(): --csv-dir (default "seeds/relatos/"), --count (int, default 10000), --force
  2. create_tables()
  3. session = SessionLocal()
  4. existing = session.query(func.count(ReportModel.id)).scalar()
     if existing >= count // 2 and not force:
         print(f"Skip: {existing} reports already in DB. Use --force to re-seed."); exit(0)

  5. # Load user_ids
     user_ids = [r[0] for r in session.query(UserModel.id).all()]
     if not user_ids: error("Run seed_users.py first"); exit(1)

  6. # Build email→id map for CSV id_cidadao resolution
     email_to_id = {r.email: r.id for r in session.query(UserModel).all()}
     citizen_fallback = user_ids[0]  # any user as fallback

  7. # Build topico→report_type_id map (case-insensitive, strip)
     type_map = {r.name.strip().lower(): r.id
                 for r in session.query(ReportTypeModel).filter_by(active=True).all()}
     if not type_map: error("Run seed_report_types.py first"); exit(1)
     default_type_id = list(type_map.values())[0]

  8. # Load CSV corpus
     corpus = load_csv_corpus(csv_dir, email_to_id, citizen_fallback, type_map, default_type_id)
     if corpus:
         print(f"Loaded {len(corpus)} rows from {csv_dir}.")
     else:
         print(f"No CSV files found in {csv_dir}. Using built-in templates.")
         corpus = build_synthetic_corpus(user_ids, type_map)

  9. # Replicate corpus with jitter to reach count
     reports = []
     for i in range(count):
         base = corpus[i % len(corpus)]
         reports.append(ReportModel(
             id=str(uuid.uuid4()),
             text=base["text"],
             lat=round(base["lat"] + random.uniform(-0.003, 0.003), 6),  # ~300m jitter
             lon=round(base["lon"] + random.uniform(-0.003, 0.003), 6),
             urgency=weighted_choice(URGENCY_WEIGHTS),
             status=weighted_choice(STATUS_WEIGHTS),
             report_type_id=base["report_type_id"],
             author_id=base["author_id"],
             photo_url=None,
             created_at=jitter_date(base["created_at"]),
         ))

  10. session.bulk_save_objects(reports)
      session.commit()
      print(f"Done. Created: {count} reports over the past 365 days.")
```

**Função `load_csv_corpus`:**

```python
def load_csv_corpus(csv_dir, email_to_id, citizen_fallback, type_map, default_type_id):
    rows = []
    csv_dir_path = Path(csv_dir)
    if not csv_dir_path.exists():
        return rows
    for csv_file in sorted(csv_dir_path.glob("*.csv")):
        with csv_file.open(encoding="utf-8") as f:
            reader = csv.DictReader(f)
            for line in reader:
                email = f"{line['id_cidadao'].strip()}@gavea.br"
                author_id = email_to_id.get(email, citizen_fallback)
                topico_key = line["topico"].strip().lower()
                type_id = type_map.get(topico_key, default_type_id)
                rows.append({
                    "text": line["texto_relato"].strip(),
                    "lat": float(line["latitude"]),
                    "lon": float(line["longitude"]),
                    "created_at": parse_date(line["data"]),
                    "report_type_id": type_id,
                    "author_id": author_id,
                })
    return rows
```

**Função `parse_date`:** aceita `YYYY-MM-DD` ou `YYYY-MM-DDTHH:MM:SS`; retorna `datetime` naive.

**Função `jitter_date`:**
```python
def jitter_date(base_dt: datetime) -> datetime:
    # Jitter ±15 days around base date, clamp to past-year window
    delta = timedelta(days=random.randint(-15, 15), hours=random.randint(0, 23))
    now = datetime.utcnow()
    year_ago = now - timedelta(days=365)
    result = base_dt + delta
    return max(year_ago, min(now, result))
```
